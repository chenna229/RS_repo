Error Volume =

VAR ChartType = 
    SELECTEDVALUE(ChartToggle[TYPE], "Count")

VAR vLatestProcessingdate = 
    MAX(SIRA_Submission_Error_Report[processing_dt])

VAR SuspectRules =
    CALCULATETABLE(
        VALUES(SIRA_Submission_Error_Report[rule_cd]),
        ALL(SIRA_Submission_Error_Report),
        SIRA_Submission_Error_Report[Rule Severity] = "Suspect"
    )

-- Get distinct AggrKeys for current Insurer only
-- NOT Insurer Team level
VAR ValidAggrKeys =
    CALCULATETABLE(
        DISTINCT(Sira_junction_table[%AggrKey]),
        ALLSELECTED(Sira_junction_table[Insurer])
    )

-- COUNT branch
VAR CountResult =
    CALCULATE(
        COUNTROWS(SIRA_Submission_Error_Report),
        SIRA_Submission_Error_Report[exclusion_type] = "N",
        SIRA_Submission_Error_Report[ErrorType] = "Proactive",
        SIRA_Submission_Error_Report[processing_dt] = vLatestProcessingdate
    )

-- DOLLAR VALUE branch
VAR DollarResult =
    CALCULATE(
        SUMX(
            VALUES(SIRA_Submission_Error_Report[source_claim_nbr]),
            CALCULATE(
                MAX(SIRA_Submission_Error_Report[pit_payment_amt]),
                SIRA_Submission_Error_Report[exclusion_type] = "N",
                SIRA_Submission_Error_Report[Rule Severity] = "Critical$",
                SIRA_Submission_Error_Report[ErrorType] = "Proactive",
                SIRA_Submission_Error_Report[processing_dt] = vLatestProcessingdate,
                NOT SIRA_Submission_Error_Report[rule_cd] IN SuspectRules
            )
        ),
        SIRA_Submission_Error_Report[%AggrKey] IN ValidAggrKeys
    )

RETURN
    IF(ChartType = "Count", CountResult, DollarResult)
