let
    // Reference original table
    Source = SIRA_Submission_Error_Report,

    // Step 1: Filter required rows
    FilteredRows = Table.SelectRows(
        Source,
        each [exclusion_type] = "N"
            and [ErrorType] = "Proactive"
            and [Rule Severity] = "Critical$"
    ),

    // Step 2: Keep only needed columns
    SelectedColumns = Table.SelectColumns(
        FilteredRows,
        {
            "source_claim_nbr",
            "processing_dt",
            "pit_payment_amt",
            "rule_cd",
            "Rule Severity"
        }
    ),

    // Step 3: Group by claim + processing_dt
    // Get MAX pit_payment_amt per claim per date
    GroupedRows = Table.Group(
        SelectedColumns,
        {
            "source_claim_nbr",
            "processing_dt"
        },
        {
            {
                "max_pit_payment",
                each List.Max([pit_payment_amt]),
                type number
            }
        }
    )

in
    GroupedRows



Claim_Payment_Aggregated



Error Volume =

VAR ChartType =
    SELECTEDVALUE(ChartToggle[TYPE], "Count")

VAR CountResult =
    CALCULATE(
        COUNTROWS(SIRA_Submission_Error_Report),
        SIRA_Submission_Error_Report[exclusion_type] = "N",
        SIRA_Submission_Error_Report[ErrorType] = "Proactive"
    )

VAR DollarResult =
    SUM(Claim_Payment_Aggregated[max_pit_payment])

RETURN
    IF(ChartType = "Count", CountResult, DollarResult)
