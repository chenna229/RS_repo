Error Value =

VAR CurrMaxProcessingDt =
    CALCULATE(
        MAX(SIRA_Submission_Error_Report[processing_dt]),
        SIRA_Submission_Error_Report[ErrorType] = "Proactive"
    )

VAR SuspectRules =
    CALCULATETABLE(
        DISTINCT(SIRA_Submission_Error_Report[rule_cd]),
        ALL(SIRA_Submission_Error_Report),
        SIRA_Submission_Error_Report[Rule Severity] = "Suspect"
    )

-- Sum of max pit_payment_amt per claim
-- Critical$ only, exclude Suspect rules
VAR ErrorValue =
    SUMX(
        VALUES(SIRA_Submission_Error_Report[source_claim_nbr]),
        CALCULATE(
            MAX(SIRA_Submission_Error_Report[pit_payment_amt]),
            SIRA_Submission_Error_Report[Rule Severity] = "Critical$",
            SIRA_Submission_Error_Report[ErrorType] = "Proactive",
            SIRA_Submission_Error_Report[exclusion_type] = "N",
            SIRA_Submission_Error_Report[processing_dt] = CurrMaxProcessingDt,
            NOT SIRA_Submission_Error_Report[rule_cd] IN SuspectRules
        )
    )

RETURN
ErrorValue
