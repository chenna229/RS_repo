SELECT
    -- Error fields from DQHUB
    dq.DQ_RULE_OCCURRENCE_ID                                        AS rule_cd,
    dq.DQ_RULE_OCCURRENCE_NAME                                      AS rule_desc,
    dq.BUSINESS_KEY_VAL                                             AS claim_number_raw,
    dq.DQ_FIELD_NAME                                                AS column_name,
    dq.DQ_FIELD_VAL                                                 AS erroneous_value,
    dq.IMPACT                                                       AS severity,
    dq.RUN_DATE                                                     AS processing_dt,
    TO_CHAR(dq.RUN_DATE, 'YYYYMM')                                  AS submission_period,

    -- Submission period from SIRA helper
    SubPeriod.SUBMISSION_PERIOD                                     AS submission_period_sira,
    SubPeriod.SUBMISSION_PERIOD_START_DT                            AS submission_period_start_dt,
    SubPeriod.SUBMISSION_PERIOD_END_DT                              AS submission_period_end_dt,

    -- Claim fields
    cc_claim."PublicID"                                             AS claim_public_id,
    cc_claim."ClaimNumber"                                          AS claim_number,
    cc_claim."SIRAClaimNumber_Ext"                                  AS sira_claim_number,
    cc_claim."LossDate"                                             AS loss_date,
    cc_claim."CloseDate"                                            AS close_date,
    cc_claim."ReOpenDate"                                           AS reopen_date,
    cc_claim."State"                                                AS state,
    cc_claim."PolicyID"                                             AS policy_id,
    cc_claim."AssignedUserID"                                       AS assigned_user_id,
    cc_claim."AssignedGroupID"                                      AS assigned_group_id,
    cc_claim."Retired"                                              AS retired,
    cc_claim."Currency"                                             AS currency,
    CAST(COALESCE(cc_claim."ClaimWorkCompID", '0') AS VARCHAR)      AS fk_claimworkcompid,

    -- Claim state description
    St."TYPECODE"                                                   AS claim_state_desc,

    -- Reopen reason
    COR."TYPECODE"                                                  AS reopen_reason_cd,

    -- Loss type
    lt."TYPECODE"                                                   AS loss_type,

    -- Reasonable excuse
    RExc."TYPECODE"                                                 AS reasonable_excuse_cd,

    -- Liability status
    ls."LiabilityStatus"                                            AS liability_status,
    ls."CTMLiabilityStatusDecisionDate"                             AS liability_status_decision_date,
    COALESCE(clt."TYPECODE", '00')                                  AS liability_status_desc,

    -- Managing entity / Insurer
    ccx_managingentity_icare."Code"                                 AS managing_entity_cd,
    ccx_managingentity_icare."Name"                                 AS managing_entity_name,
    CAST(COALESCE(MAG."TYPECODE", LAG."TYPECODE") AS VARCHAR(5))    AS insurer_number,
    IC."INSURER_NUMBER"                                             AS insurer_control_number,

    -- Policy fields
    cc_policy."PolicyNumber"                                        AS policy_number,
    cc_policy."PolicyType"                                          AS policy_type_id,
    cctl_policytype."TYPECODE"                                      AS policy_type_cd,

    -- Case owner / Team
    pia."CaseOwnerName"                                             AS claims_officer,
    pia."CaseOwnerTeamName"                                         AS team_name,

    -- Work status (via workcomp)
    WCS."STATUS_CD"                                                 AS work_status_cd,

    -- Transaction fields
    txn."ID"                                                        AS transaction_id,
    txn."PublicID"                                                  AS transaction_public_id,
    txn."PaymentType"                                               AS payment_type,
    txn."SubmitDate"                                                AS submit_date,
    txn."Status"                                                    AS transaction_status,
    txn."CreateTime"                                                AS transaction_create_time,
    txn."UpdateTime"                                                AS transaction_update_time,

    -- Transaction line item
    tli."ID"                                                        AS line_item_id,
    tli."TransactionAmount"                                         AS transaction_amount,

    -- Payment fields
    pay.COSTCATEGORYCODE                                            AS cost_category_cd,
    pay.COSTCATEGORYNAME                                            AS cost_category_name,
    pay.TRANSACTIONTYPE                                             AS payment_transaction_type,
    pay.TRANSACTIONSUBTYPE                                          AS payment_transaction_subtype,
    pay.TRANSACTIONDATEYYYYMM                                       AS transaction_date_yyyymm,
    pay.TRANSACTIONAMOUNT                                           AS pit_payment_amt,
    pay.STATUS_CD                                                   AS payment_status_cd,
    pay.ODGRTWDATE_ICARE                                            AS odgrtw_date,
    pay.ODGDURATION_ICARE                                           AS odg_duration,

    -- Recovery fields
    rec.TRANSACTIONAMOUNT                                           AS recovery_amt,
    rec.TRANSACTIONDATE                                             AS recovery_transaction_date,
    rec.PAYCODE                                                     AS recovery_paycode,
    rec.STATUSCD                                                    AS recovery_status_cd,
    rec.STATUSNAME                                                  AS recovery_status_name

FROM DATALAB_CDQ_HUB_DEV.D253_DQU."v_DQHUB_INVALID_DATA" dq

-- Join to claim
LEFT JOIN DATALAB_DATA_MIGRATION_DEV.D111_DQ_GWCC."v_cc_claim_current" cc_claim
    ON cc_claim."PublicID" = dq.BUSINESS_KEY_VAL
    AND cc_claim."Retired" = 0

-- Submission period (1=1 as only 1 active row)
LEFT JOIN (
    SELECT SUBMISSION_PERIOD, SUBMISSION_PERIOD_END_DT, SUBMISSION_PERIOD_START_DT
    FROM DATALAB_SIRA_PROD.P001_HELPER.V_SIRA_SUBMISSION_PERIOD
    WHERE CURRENT_SUBMISSION_FLAG = 'Y'
) SubPeriod ON 1=1

-- Case owner name and team
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."ccx_piaweindexremreport_ext" pia
    ON pia."ID" = cc_claim."ID"
    AND pia."CURRENT_RECORD_FLAG" = TRUE

-- Managing entity
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."ccx_managingentity_icare" ccx_managingentity_icare
    ON ccx_managingentity_icare."ID" = cc_claim."ManagingEntity_icare"
    AND ccx_managingentity_icare."CURRENT_RECORD_FLAG" = TRUE

-- Claims agent primary (LAG - lodging agent)
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_claimagent_icare_current" LAG
    ON cc_claim."LodgingAgent_icare" = LAG."ID"

-- Claims agent (MAG - claims agent)
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_claimagent_icare_current" MAG
    ON cc_claim."ClaimsAgent_icare" = MAG."ID"

-- Insurer control (for insurer name)
LEFT JOIN DATALAB_SIRA_PROD.P001_CONSUMPTION."INSURER_CONTROL" IC
    ON IC."INSURER_NUMBER" = MAG."TYPECODE"
    AND IC."ACTIVE_IND" = 'Y'

-- Policy
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."cc_policy" cc_policy
    ON cc_policy."ID" = cc_claim."PolicyID"
    AND cc_policy."CURRENT_RECORD_FLAG" = TRUE

-- Policy type
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."cctl_policytype" cctl_policytype
    ON cctl_policytype."ID" = cc_policy."PolicyType"
    AND cctl_policytype."CURRENT_RECORD_FLAG" = TRUE

-- Claim state description
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_claimstate_current" St
    ON St."ID" = cc_claim."State"
    AND St."Retired" = 0

-- Reopen reason
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_claimreopenedreason_current" COR
    ON COR."ID" = cc_claim."ReopenedReason"
    AND COR."Retired" = 0

-- Loss type
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_losstype_current" lt
    ON lt."ID" = cc_claim."LossType"

-- Reasonable excuse (via workcomp)
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_reasonableexcuse_icare_current" RExc
    ON RExc."ID" = WCS."ReasonableExcuse_icare"

-- Liability status (latest per claim)
LEFT JOIN (
    SELECT "ClaimWorkCompID", "CTMLiabilityStatusDecisionDate", "LiabilityStatus", "Run_id"
    FROM (
        SELECT tl.*,
               ROW_NUMBER() OVER (
                   PARTITION BY "ClaimWorkCompID"
                   ORDER BY "CTMLiabilityStatusDecisionDate" DESC
               ) AS row_num
        FROM DATALAKE_DATA_MIGRATION_DEV.D111_DQ_GWCC."v_ccx_liabilitystatushist_icare_current" tl
        WHERE "Retired" = 0
        AND "CTMLiabilityStatusDecisionDate" IS NOT NULL
        AND "LiabilityStatus" IS NOT NULL
    ) WHERE row_num = 1
) ls
    ON ls."ClaimWorkCompID" = cc_claim."ClaimWorkCompID"
    AND ls."Run_id" = cc_claim."Run_id"

-- Liability status description
LEFT JOIN DATALAKE_CURATED_PROD.P001_GWCC."v_cctl_compensabilitydecision_current" clt
    ON clt."ID" = ls."LiabilityStatus"

-- Workcomp (for work status)
LEFT JOIN (
    SELECT * FROM DATALAB_DATA_MIGRATION_DEV.D111_DQ_GWCC."v_cc_workcomp_current"
    WHERE "Retired" = 0
) WCS
    ON WCS."ID" = cc_claim."ClaimWorkCompID"
    AND WCS."Run_id" = cc_claim."Run_id"

-- Transaction
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."cc_transaction" txn
    ON txn."ClaimID" = cc_claim."ID"
    AND txn."CURRENT_RECORD_FLAG" = TRUE

-- Transaction line item
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."cc_transactionlineitem" tli
    ON tli."TransactionID" = txn."ID"
    AND tli."CURRENT_RECORD_FLAG" = TRUE

-- Payments
LEFT JOIN DATALAKE_CURATED_DEV.D111_DATA_MIGRATION_AAL."v_sira_c_2_6_estimates_payments" pay
    ON pay.CLAIMNUMBER = cc_claim."ClaimNumber"

-- Recoveries
LEFT JOIN DATALAKE_CURATED_DEV.D111_DATA_MIGRATION_AAL."v_sira_c_2_5_recoveries" rec
    ON rec.CLAIMNUMBER = cc_claim."ClaimNumber"

WHERE dq.TABLE_NAME = 'cc_claim'
AND dq.BUSINESS_KEY = 'PublicID'
AND ccx_managingentity_icare."Code" LIKE 'NI_%'

LIMIT 100;
