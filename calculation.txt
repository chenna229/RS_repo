Error Rate =

VAR CurrentPeriodKey =
    MAX(SIRA_Submission_Error_Report[Submission_Period_Key])

VAR PrevKeyText =
    FORMAT(
        EDATE(
            DATE(
                VALUE(LEFT(CurrentPeriodKey, 4)),
                VALUE(MID(CurrentPeriodKey, 5, 2)),
                1
            ),
        -1),
    "YYYYMMDD"
    )

VAR PrevMaxProcessingDt =
    CALCULATE(
        MAX(SIRA_Submission_Error_Report[processing_dt]),
        REMOVEFILTERS(Calendar),
        SIRA_Submission_Error_Report[Submission_Period_Key] = PrevKeyText,
        SIRA_Submission_Error_Report[ErrorType] = "Proactive"
    )

VAR CurrMaxProcessingDt =
    CALCULATE(
        MAX(SIRA_Submission_Error_Report[processing_dt]),
        SIRA_Submission_Error_Report[ErrorType] = "Proactive"
    )

VAR A =
    CALCULATE(
        DISTINCTCOUNT(SIRA_Submission_Error_Report[ClaimandErrorCode]),
        REMOVEFILTERS(Calendar),
        SIRA_Submission_Error_Report[Submission_Period_Key] = PrevKeyText,
        SIRA_Submission_Error_Report[processing_dt] = PrevMaxProcessingDt,
        SIRA_Submission_Error_Report[Rule Severity] = "Critical$",
        SIRA_Submission_Error_Report[ErrorType] = "Proactive",
        SIRA_Submission_Error_Report[exclusion_type] = "N"
    )

VAR PrevPeriodCodes =
    CALCULATETABLE(
        VALUES(SIRA_Submission_Error_Report[ClaimandErrorCode]),
        REMOVEFILTERS(Calendar),
        SIRA_Submission_Error_Report[Submission_Period_Key] = PrevKeyText,
        SIRA_Submission_Error_Report[processing_dt] = PrevMaxProcessingDt,
        SIRA_Submission_Error_Report[ErrorType] = "Proactive"
    )

VAR B =
    CALCULATE(
        DISTINCTCOUNT(SIRA_Submission_Error_Report[ClaimandErrorCode]),
        SIRA_Submission_Error_Report[ClaimandErrorCode] IN PrevPeriodCodes,
        SIRA_Submission_Error_Report[processing_dt] = CurrMaxProcessingDt,
        SIRA_Submission_Error_Report[Rule Severity] = "Critical$",
        SIRA_Submission_Error_Report[ErrorType] = "Proactive",
        -- Handle both Text and Number
        VALUE(SIRA_Submission_Error_Report[error_age_months]) > 0,
        SIRA_Submission_Error_Report[exclusion_type] = "N"
    )

VAR C = A

RETURN
DIVIDE(A - B, C)
