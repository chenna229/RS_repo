-- Is %AggrKey unique in error report?
Test_AggrKey_Unique =
COUNTROWS(SIRA_Submission_Error_Report)
-
DISTINCTCOUNT(SIRA_Submission_Error_Report[%AggrKey])



-- How many rows share same %AggrKey?
Test_AggrKey_Dupes =
AVERAGEX(
    VALUES(SIRA_Submission_Error_Report[%AggrKey]),
    CALCULATE(
        COUNTROWS(SIRA_Submission_Error_Report)
    )
)


-- How many distinct %AggrKey in junction for EML?
Test_Junction_EML_Keys =
CALCULATE(
    DISTINCTCOUNT(Sira_junction_table[%AggrKey]),
    Sira_junction_table[Insurer] = "EML"
)
