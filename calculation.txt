SELECT
    -- Claim fields
    dt_cc_claim_current."ClaimNumber"
    , dt_cc_claim_current."MANAGING_ENTITY_CD"
    , dt_cc_claim_current."MANAGINGENTITY_NAME"
    , dt_cc_claim_current."INSURERNUMBER"
    , dt_cc_claim_current."POLICYNUMBER"
    , dt_cc_claim_current."POLICYTYPE_ID"
    , dt_cc_claim_current."POLICYTYPE_CD"

    -- Transaction fields
    , cc_transaction."ID"                   AS transaction_id
    , cc_transaction."PublicID"             AS transaction_public_id
    , cc_transaction."PaymentType"
    , cc_transaction."SubmitDate"
    , cc_transaction."CloseClaim"
    , cc_transaction."Status"
    , cc_transaction."CreateTime"
    , cc_transaction."UpdateTime"

    -- Transaction line item fields
    , tli."ID"                              AS line_item_id
    , tli."TransactionAmount"
    , tli."Amount"
    , tli."Currency"

FROM DATALAKE_CURATED_DEV.D253_GWCC."cc_transaction" AS cc_transaction

LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."dt_cc_claim_current" AS dt_cc_claim_current
    ON dt_cc_claim_current."ID" = cc_transaction."ClaimID"

LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."cc_transactionlineitem" AS tli
    ON tli."TransactionID" = cc_transaction."ID"
    AND tli.CURRENT_RECORD_FLAG = TRUE

WHERE cc_transaction.CURRENT_RECORD_FLAG = TRUE;
