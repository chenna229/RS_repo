Find reopen reason table
SELECT TABLE_NAME, COLUMN_NAME
FROM DATALAKE_CURATED_DEV.INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'D253_GWCC'
AND (
    UPPER(TABLE_NAME) LIKE '%REOPEN%'
    OR UPPER(TABLE_NAME) LIKE '%OPENREASON%'
)
ORDER BY TABLE_NAME, COLUMN_NAME;

-- Find work status / status column on workcomp
SELECT TABLE_NAME, COLUMN_NAME
FROM DATALAKE_CURATED_DEV.INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'D253_GWCC'
AND UPPER(TABLE_NAME) LIKE '%WORKCOMP%'
ORDER BY TABLE_NAME, COLUMN_NAME;
