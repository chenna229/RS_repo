SELECT TOP 10
    c."ClaimNumber",
    p."CaseOwnerName",
    p."CaseOwnerTeamName"
FROM DATALAKE_CURATED_DEV.D253_GWCC."cc_claim" c
LEFT JOIN DATALAKE_CURATED_DEV.D253_GWCC."ccx_piaweindexremreport_ext" p
    ON p."ID" = c."ID"
    AND p."CURRENT_RECORD_FLAG" = TRUE
WHERE c."CURRENT_RECORD_FLAG" = TRUE
LIMIT 10;
