Test_A_Fix2 =

VAR SuspectRules =
    CALCULATETABLE(
        VALUES(SIRA_Submission_Error_Report[rule_cd]),
        SIRA_Submission_Error_Report[Rule Severity] = "Suspect"
    )

VAR BaseTable =
    CALCULATETABLE(
        SIRA_Submission_Error_Report,
        SIRA_Submission_Error_Report[Rule Severity] = "Critical$",
        SIRA_Submission_Error_Report[ErrorType] = "Proactive",
        SIRA_Submission_Error_Report[exclusion_type] = "N",
        NOT SIRA_Submission_Error_Report[rule_cd] IN SuspectRules
    )

VAR A =
    SUMX(
        SUMMARIZE(
            BaseTable,
            SIRA_Submission_Error_Report[source_claim_nbr],
            SIRA_Submission_Error_Report[processing_dt]
        ),
        MAXX(
            FILTER(
                BaseTable,
                SIRA_Submission_Error_Report[source_claim_nbr] = EARLIER(SIRA_Submission_Error_Report[source_claim_nbr])
                && SIRA_Submission_Error_Report[processing_dt] = EARLIER(SIRA_Submission_Error_Report[processing_dt])
            ),
            SIRA_Submission_Error_Report[pit_payment_amt]
        )
    )

RETURN A
