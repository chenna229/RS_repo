-- Test previous period key
Test_PrevKey =
VAR CurrentPeriodKey =
    MAX(SIRA_Submission_Error_Report[Submission_Period_Key])
VAR PreviousPeriod =
    VALUE(
        FORMAT(
            EDATE(
                DATE(
                    LEFT(FORMAT(CurrentPeriodKey,"00000000"),4),
                    VALUE(MID(FORMAT(CurrentPeriodKey,"00000000"),5,2)),
                    1
                ),
            -1),
        "YYYYMMDD"
        )
    )
RETURN PreviousPeriod
