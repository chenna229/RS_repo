Overall Error Rate% =

VAR SuspectRules =
    CALCULATETABLE(
        VALUES(SIRA_Submission_Error_Report[rule_cd]),
        ALL(SIRA_Submission_Error_Report),
        SIRA_Submission_Error_Report[Rule Severity] = "Suspect"
    )

VAR CurrMaxProcessingDt =
    CALCULATE(
        MAX(SIRA_Submission_Error_Report[processing_dt]),
        SIRA_Submission_Error_Report[ErrorType] = "Proactive"
    )

-- Numerator
VAR Numerator =
    CALCULATE(
        SUMX(
            VALUES(SIRA_Submission_Error_Report[source_claim_nbr]),
            CALCULATE(
                MAX(SIRA_Submission_Error_Report[pit_payment_amt]),
                SIRA_Submission_Error_Report[Rule Severity] = "Critical$",
                SIRA_Submission_Error_Report[exclusion_type] = "N",
                SIRA_Submission_Error_Report[ErrorType] = "Proactive",
                SIRA_Submission_Error_Report[processing_dt] = CurrMaxProcessingDt,
                NOT SIRA_Submission_Error_Report[rule_cd] IN SuspectRules
            )
        )
    )

-- Denominator
-- PROCESSING_DT in AGGREGATE_MEASURES_SIRA is Date type
VAR Denominator =
    CALCULATE(
        SUMX(
            SUMMARIZE(
                AGGREGATE_MEASURES_SIRA,
                AGGREGATE_MEASURES_SIRA[DIMENSION_1_TXT],
                AGGREGATE_MEASURES_SIRA[DIMENSION_2_TXT]
            ),
            CALCULATE(
                MAX(AGGREGATE_MEASURES_SIRA[paymentAmt_PIT_Total]),
                -- Compare Date to Date directly
                AGGREGATE_MEASURES_SIRA[PROCESSING_DT] = CurrMaxProcessingDt
            )
        ),
        REMOVEFILTERS(SIRA_Submission_Error_Report[Rule Severity]),
        REMOVEFILTERS(SIRA_Submission_Error_Report[rule_cd])
    )

RETURN
DIVIDE(Numerator, Denominator)
