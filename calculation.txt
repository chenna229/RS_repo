let
    Source = SIRA_Submission_Error_Report,

    // Without Rule Severity filter
    FilteredRows = Table.SelectRows(
        Source,
        each [exclusion_type] = "N"
            and [ErrorType] = "Proactive"
            and [submission_period] = 202603
            and [processing_dt] = #date(2026, 4, 4)
    ),

    // Count distinct claims WITH Critical$
    CriticalClaims = List.Distinct(
        Table.SelectRows(
            FilteredRows,
            each [Rule Severity] = "Critical$"
        )[source_claim_nbr]
    ),

    // Count distinct claims WITHOUT Rule Severity filter
    AllClaims = List.Distinct(
        FilteredRows[source_claim_nbr]
    ),

    Result = #table(
        {"Critical$ Claims", "All Claims"},
        {{List.Count(CriticalClaims), List.Count(AllClaims)}}
    )

in
    Result
