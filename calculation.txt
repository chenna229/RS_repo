SELECT 
    TABLE_SCHEMA,
    TABLE_NAME,
    COLUMN_NAME,
    DATA_TYPE
FROM DATALAKE_CURATED_DEV.INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'D253_GWCC'
  AND (
    UPPER(COLUMN_NAME) LIKE '%OFFICER%'
    OR UPPER(COLUMN_NAME) LIKE '%MANAGER%'
    OR UPPER(COLUMN_NAME) LIKE '%ASSIGN%'
    OR UPPER(COLUMN_NAME) LIKE '%OWNER%'
    OR UPPER(COLUMN_NAME) LIKE '%HANDLER%'
    OR UPPER(COLUMN_NAME) LIKE '%USER%'
  )
ORDER BY TABLE_NAME, COLUMN_NAME;
