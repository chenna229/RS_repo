SELECT TABLE_NAME, COLUMN_NAME
FROM DATALAKE_CURATED_DEV.INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'D253_GWCC'
AND (
    UPPER(TABLE_NAME) LIKE '%INSURER%'
    OR UPPER(TABLE_NAME) LIKE '%INSURER_CONTROL%'
    OR UPPER(COLUMN_NAME) LIKE '%INSURER_NUMBER%'
    OR UPPER(COLUMN_NAME) LIKE '%ACTIVE_IND%'
)
ORDER BY TABLE_NAME, COLUMN_NAME;


SELECT TABLE_NAME, COLUMN_NAME
FROM DATALAB_CDQ_HUB_DEV.INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'D253_DQU'
AND (
    UPPER(TABLE_NAME) LIKE '%INSURER%'
    OR UPPER(COLUMN_NAME) LIKE '%INSURER%'
)
ORDER BY TABLE_NAME, COLUMN_NAME;
